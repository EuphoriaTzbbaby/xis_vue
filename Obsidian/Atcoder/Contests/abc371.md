auto solve = [] () {
	ll n;

	cin >> n;

	vector<ll> nums(n);

	fw(i, 0, n) cin >> nums[i];

	ll q;

	cin >> q;

	ll ans = 0;

	while(q--) {

		ll id, pos;

		cin >> id >> pos;

		id -= 1;

		ll t = nums[id];

		nums[id] = pos;

		ans += abs(pos - t);

		ll d = 0;

		if(pos >= t) {

			for(ll j = id + 1; j < n; j += 1) {

				if(nums[j] <= pos) {

					ans += (pos + d + 1) - nums[j];

					nums[j] = pos + (++d);

				}

			}

		} else {

			for(ll j = id - 1; j >= 0; j -= 1) {

				if(nums[j] >= pos) {

					ans += nums[j] - (pos - (d + 1));

					nums[j] = pos - (++d);

				}

			}

		}

	}

	cout << ans << endl;

};
